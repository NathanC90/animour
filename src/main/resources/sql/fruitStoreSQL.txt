drop table fruitstore;

create table FruitStore
(
    id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    storeName varchar(50),
    address varchar(50),
    phoneNumber varchar(20),
    businessHours time
);

drop table fruit;

create table fruit
(
    fruit_id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    product_id bigint references fruitstore(id),
    fruitName varchar(20),
    price integer,
    psource varchar(30)
)
